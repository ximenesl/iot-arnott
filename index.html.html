4 setup() — inicialização única Executa 1x ao ligar ▼ 
/*
 * ================================================================
 *  BLOCO 4 — SETUP (INICIALIZACAO)
 *  --------------------------------------------------
 *  A funcao setup() roda UMA UNICA VEZ quando a placa liga.
 *  Ela prepara o ambiente: liga a porta Serial, configura os
 *  pinos (entrada ou saida) e inicializa o WiFi e o sensor.
 * ================================================================
 */
void setup() {
  // 1. Inicia a comunicacao Serial (para o Monitor Serial do PC)
  // IMPORTANTE: Configure seu Monitor Serial para 115200 baud!
  Serial.begin(115200);
  delay(500);  // Pausa de meio segundo para estabilizar a conexao
  
  Serial.println("\n================================================");
  Serial.println("  Sistema IoT SENAC v3.0 — Iniciando...");
  Serial.println("================================================");

  // 2. Configura o pino do LED
  // OUTPUT: O ESP32 manda energia para o pino (acende o LED)
  pinMode(PINO_LED, OUTPUT);
  digitalWrite(PINO_LED, LOW); // Garante que o LED comece apagado
  Serial.println("[OK] Pinos de hardware configurados.");

  // 3. Acorda o sensor DHT
  dht.begin();
  Serial.println("[OK] Sensor DHT inicializado.");

  // 4. Conecta na rede WiFi (usando a funcao que criamos)
  conectarWiFi();

  Serial.println("[OK] Setup concluido. Iniciando ciclo continuo...\n");
}
5 loop() — ciclo principal com millis() Executa infinitamente ▼ 
/*
 * ================================================================
 *  BLOCO 5 — LOOP PRINCIPAL (O CORACAO DO PROGRAMA)
 *  --------------------------------------------------
 *  A funcao loop() roda REPETIDAS VEZES, sem parar, enquanto a 
 *  placa tiver energia.
 *  
 *  DICA DE OURO: Evite usar delay(10000)!
 *  O delay() "trava" o processador. Se a placa estiver travada,
 *  ela nao percebe um botao sendo apertado, por exemplo.
 *  Usamos o millis() (cronometro da placa) para agendar tarefas.
 * ================================================================
 */
void loop() {
  // Pega o tempo atual do cronometro interno do ESP32 (em milissegundos)
  unsigned long agora = millis();

  // Verifica se o tempo que passou desde a ultima execucao e maior que o nosso limite
  if (agora - ultimaExec >= INTERVALO_MS) {
    
    // Atualiza a ultima execucao para "agora", resetando o nosso cronometro
    ultimaExec = agora;
    Serial.println("--- Novo ciclo de leitura e envio ---");

    // Passo 1: Ler o sensor
    // Se a leitura falhar (retornar false), nos abortamos o envio atual usando "return;"
    if (!lerSensor()) {
      Serial.println("[ERRO] Leitura falhou. Abortando este ciclo.");
      return; 
    }

    // Passo 2: Executar a regra de negocio (Atuar no mundo fisico)
    acionarLED(tempAtual);

    // Passo 3: Garantir que a internet continua funcionando
    verificarWiFi();

    // Passo 4: Enviar os dados para o seu banco de dados na Nuvem/API
    if (WiFi.status() == WL_CONNECTED) {
      bool enviado = enviarParaAPI(tempAtual, umidAtual);
      if (enviado) {
        Serial.println("[OK] Dados enviados para a API com sucesso.");
      } else {
        Serial.println("[AVISO] Falha ao enviar para a API. Tentaremos no proximo ciclo.");
      }
    } else {
      Serial.println("[AVISO] Sem internet. Nao e possivel enviar agora.");
    }
  }

  /*
   * ==============================================================
   *  ZONA LIVRE - AREA MULTITAREFA
   *  O processador passa por aqui milhares de vezes por segundo!
   *  Voce pode colocar codigos aqui para ler botoes ou controlar 
   *  um display, pois o codigo nao esta preso em delays.
   * ==============================================================
   */
}
6 conectarWiFi() e verificarWiFi() Gestao de conectividade ▼ 
/*
 * ================================================================
 *  BLOCO 6 — FUNCOES DE INTERNET (WIFI)
 *  --------------------------------------------------
 *  Duas funcoes essenciais para projetos IoT reais:
 *  - conectarWiFi(): Chamada ao ligar, tenta conectar pela 1a vez.
 *  - verificarWiFi(): Chamada sempre. Se a internet cair, ela 
 *    reconecta sozinha. Essencial para aparelhos que ficam ligados
 *    na parede o ano todo sem ninguem monitorando.
 * ================================================================
 */
void conectarWiFi() {
  Serial.print("[WiFi] Conectando a rede: ");
  Serial.println(WIFI_SSID);

  WiFi.mode(WIFI_STA); // Define a placa como "Estacao" (Cliente de um roteador)
  WiFi.begin(WIFI_SSID, WIFI_SENHA);

  int tentativas = 0;
  // Tenta conectar, piscando o LED enquanto carrega
  while (WiFi.status() != WL_CONNECTED && tentativas < 20) {
    digitalWrite(PINO_LED, !digitalRead(PINO_LED)); // Inverte o estado do LED (Pisca)
    delay(500);
    Serial.print(".");
    tentativas++;
  }
  Serial.println();

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("[WiFi] Sucesso! Estamos online.");
    Serial.print("[WiFi] Endereco IP local: ");
    Serial.println(WiFi.localIP());
    digitalWrite(PINO_LED, LOW); // Garante que o LED apaga apos conectar
  } else {
    Serial.println("[WiFi] FALHA. Verifique o SSID ou a Senha configurados no inicio do codigo.");
  }
}

// Funcao "Salva-vidas" - Se a rede cair, o aparelho nao precisa ser reiniciado
void verificarWiFi() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[WiFi] A conexao caiu! Tentando reconectar...");
    WiFi.disconnect();
    delay(1000);
    conectarWiFi();
  }
}
7 lerSensor() — DHT22 com validação dupla isnan + range fisico ▼ 
/*
 * ================================================================
 *  BLOCO 7 — LEITURA DO SENSOR E VALIDACAO
 *  --------------------------------------------------
 *  Nunca confie cegamente nos sensores! Fios soltam, componentes 
 *  queimam. Aqui usamos o conceito de "Validacao de Dados".
 * ================================================================
 */
bool lerSensor() {
  // Pede para o DHT ler a temperatura (C) e a umidade (%)
  float t = dht.readTemperature(); 
  float u = dht.readHumidity();    

  // Validacao 1: isnan() significa "Is Not a Number".
  // Isso acontece se o fio de dados do sensor estiver desconectado.
  if (isnan(t) || isnan(u)) {
    Serial.println("[SENSOR] ERRO: O sensor nao esta respondendo. O fio de dados esta ligado no pino certo?");
    return false; // Retorna falso, abortando a leitura
  }

  // Validacao 2: Limites fisicos do Planeta Terra para o DHT22.
  // Se ele ler 200 graus, com certeza tem algo quebrado.
  if (t < -40.0 || t > 80.0 || u < 0.0 || u > 100.0) {
    Serial.println("[SENSOR] ERRO: Valor fora da realidade. Sensor pode estar danificado.");
    Serial.printf("         Lidos: Temp=%.2f Umidade=%.2f\n", t, u);
    return false;
  }

  // Se passou pelas validacoes, nos guardamos os valores nas variaveis globais
  tempAtual = t;
  umidAtual = u;

  Serial.printf("[SENSOR] Lidos com sucesso -> Temp: %.1f oC | Umidade: %.1f%%\n", tempAtual, umidAtual);
  return true; // Sucesso!
}
8 acionarLED() — atuador com controle de estado Edge computing local ▼ 
/*
 * ================================================================
 *  BLOCO 8 — ATUADOR (AGINDO NO MUNDO FISICO)
 *  --------------------------------------------------
 *  Neste projeto usamos um LED, mas na vida real este codigo
 *  acionaria um Rele para ligar um ventilador, uma bomba de agua
 *  ou um alarme. Isso e chamado de Edge Computing (A placa toma a
 *  decisao localmente).
 * ================================================================
 */
void acionarLED(float temperatura) {
  
  // Pergunta logica: A temperatura atual ultrapassou o nosso limite?
  bool deveAcender = (temperatura > TEMP_ALERTA);

  // So vamos mudar o estado fisico se for diferente de como o LED ja esta!
  // Isso evita ficar mandando a mesma mensagem pro monitor serial varias vezes.
  if (deveAcender && !ledLigado) {
    digitalWrite(PINO_LED, HIGH); // Manda energia e acende o LED
    ledLigado = true;
    Serial.printf("[LED] ALERTA! LED LIGADO — (%.1f oC passou do limite de %.1f oC)\n", temperatura, TEMP_ALERTA);

  } else if (!deveAcender && ledLigado) {
    digitalWrite(PINO_LED, LOW);  // Corta a energia e apaga o LED
    ledLigado = false;
    Serial.printf("[LED] NORMAL. LED APAGADO — (%.1f oC voltou para abaixo de %.1f oC)\n", temperatura, TEMP_ALERTA);
  }
}
9 enviarParaAPI() — HTTP POST completo JSON + tratamento de resposta ▼ 
/*
 * ================================================================
 *  BLOCO 9 — ENVIO PARA A API (A NUVEM)
 *  --------------------------------------------------
 *  E aqui que a placa deixa de ser apenas "eletronica" e 
 *  passa a ser "Internet das Coisas".
 *  Montamos um pacote de dados (JSON) e fazemos um envio 
 *  via HTTP POST para o nosso servidor.
 * ================================================================
 */
bool enviarParaAPI(float temperatura, float umidade) {

  // 1. Cria o objeto JSON com o tamanho necessario
  StaticJsonDocument<256> doc;
  
  // 2. Preenche os "pares" de Chave: Valor
  // Usamos a variavel ID_DISPOSITIVO que o aluno configurou la no inicio!
  doc["dispositivo"]  = ID_DISPOSITIVO;
  doc["temperatura"]  = temperatura;
  doc["umidade"]      = umidade;
  doc["led_ativo"]    = ledLigado;
  doc["timestamp_ms"] = millis();

  // Transforma o objeto num texto puro (String) para poder viajar pela internet
  String pacoteJSON;
  serializeJson(doc, pacoteJSON); 
  Serial.println("[API] Enviando pacote: " + pacoteJSON);

  // 3. Prepara o Carteiro (HTTP) para enviar o pacote
  HTTPClient http;
  http.setTimeout(8000); // Se o servidor demorar mais que 8 seg, desiste
  http.begin(API_URL);   // Coloca o endereco do servidor
  
  // Avisa o servidor que a linguagem que estamos falando eh JSON
  http.addHeader("Content-Type",  "application/json"); 
  // Coloca a "senha" de acesso (Token)
  http.addHeader("Authorization", API_TOKEN);           

  // 4. Manda apertar o "Enviar" do POST e pega o codigo de resposta do servidor
  int codigoRespostaHTTP = http.POST(pacoteJSON);

  bool sucesso = false;

  // Analisa o recibo que o servidor retornou
  if (codigoRespostaHTTP > 0) {
    Serial.printf("[API] Servidor respondeu: HTTP %d\n", codigoRespostaHTTP);

    // Codigo 200 (OK) ou 201 (Created) significam sucesso!
    if (codigoRespostaHTTP == HTTP_CODE_OK || codigoRespostaHTTP == HTTP_CODE_CREATED) {
      String mensagemDoServidor = http.getString();
      Serial.println("[API] Sucesso! Servidor disse: " + mensagemDoServidor);
      sucesso = true;

    } else if (codigoRespostaHTTP == 400) {
      Serial.println("[API] Erro 400: Servidor achou nosso JSON mal formado.");
    } else if (codigoRespostaHTTP == 401) {
      Serial.println("[API] Erro 401: Acesso Negado! Verifique seu API_TOKEN.");
    } else if (codigoRespostaHTTP == 404) {
      Serial.println("[API] Erro 404: Endereco não encontrado. Verifique API_URL.");
    } else {
      Serial.println("[API] Erro desconhecido.");
    }

  } else {
    // Se for negativo, nao achou o servidor na rede
    Serial.printf("[API] Falha de conexao! Servidor desligado? Erro: %s\n",
                   http.errorToString(codigoRespostaHTTP).c_str());
  }

  // 5. REGRA DE OURO: Desligue o telefone!
  // Se não dermos o http.end(), a placa trava por vazamento de memoria.
  http.end();

  return sucesso;
}

Bloco 1 — Includes
Bibliotecas sao modulos prontos. Instale pelo Library Manager da IDE: DHT sensor library (Adafruit) e ArduinoJson (Blanchon). WiFi.h e HTTPClient.h ja vem com o pacote ESP32.
Bloco 2 — Configurações do Aluno
Aqui estao as configuracoes principais! Preencha ID_DISPOSITIVO, WIFI_SSID e sua API_TOKEN. Usamos const para garantir que esses valores nao mudem acidentalmente.
Bloco 3 — Prototipos
C/C++ le de cima para baixo. Declarar assinaturas no topo evita erros de "funcao nao declarada". Tambem serve como sumario do sketch — o aluno ve todas as funcoes de uma vez.
Bloco 4 — setup()
Roda uma unica vez. Ordem importa: Serial.begin() primeiro para ver os logs, depois pinos, sensor e por ultimo WiFi. Inverter a ordem e nao ver as mensagens de erro.
Bloco 5 — millis()
delay() congela o processador. millis() e o relogio interno: compare agora - ultimaExec com o intervalo. O ESP32 fica livre para botoes, display e outros atuadores entre os ciclos.
Bloco 6 — WiFi
verificarWiFi() a cada ciclo garante reconexao automatica sem reset manual. Dispositivos IoT operam sem supervisao — resiliencia e obrigatoria em producao.
Bloco 7 — lerSensor()
Validacao dupla: isnan() detecta falha de hardware; checagem de range detecta valores fisicamente impossiveis. Nunca envie dados invalidos para a API — isso corrompe o banco de dados.
Bloco 8 — acionarLED()
Variavel ledLigado controla o estado anterior — evita spam no Serial. Substitua o LED por rele para cargas reais. Esta logica local e chamada de edge computing.
Bloco 9 — enviarParaAPI()
HTTP POST com JSON. Tratamento discriminado por codigo (200, 400, 401, 404, 500, negativo). http.end() e obrigatorio ao final — omitir causa vazamento de memoria e travamentos.
Roteiro de depuracao — Monitor Serial a 115200 baud Siga esta ordem e resolva um bloco de cada vez. Nunca tente depurar tudo simultaneamente.

Passo a passo no Monitor Serial

1 Aparece "Setup concluido"? Se nao, o problema esta no bloco 4 (setup). Verifique as conexoes fisicas.
2 Aparece o endereco IP do WiFi? Se nao, verifique WIFI_SSID e WIFI_SENHA no bloco 2.
3 Aparecem os valores do sensor? Se nao, confira o pino PINO_DHT e o modelo TIPO_DHT. Verifique a conexao fisica do sensor.
4 O LED acende acima de 28°C? Se nao, reveja a logica do bloco 8 e o limiar TEMP_ALERTA.
5 A API responde HTTP 200 ou 201? Se nao, verifique API_URL, API_TOKEN e se o servidor esta acessivel na rede.
